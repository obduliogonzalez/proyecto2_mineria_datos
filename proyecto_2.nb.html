# Funcin para entrenar y graficar un rbol de decisin
entrenar_arbol <- function(data, target, predictors, caso) {
  # Crear frmula del modelo
  formula <- as.formula(paste(target, "~", paste(predictors, collapse = "+")))
  
  # Entrenar el rbol de decisin
  modelo <- rpart(formula, data = data, method = "class")
  
  # Mostrar el resumen del modelo
  cat("\nResumen del rbol de decisin para el Caso", caso, "\n")
  print(summary(modelo))
  
  # Graficar el rbol
  rpart.plot(modelo, main = paste("rbol de decisin - Caso", caso), type = 3, extra = 102)
  
  return(modelo)
}
